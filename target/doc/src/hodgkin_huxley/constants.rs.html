1//! Physical and biological constants for the Hodgkin-Huxley model.
2//!
3//! This module contains fundamental constants used in neuronal modeling,
4//! including thermodynamic constants and physiological ion concentrations.

6/// Physical constants used in the Hodgkin-Huxley model.
7#[derive(Debug, Clone, Copy)]
8pub struct PhysicalConstants {
9    /// Faraday constant (C/mol)
10    pub faraday: f64,
11    /// Universal gas constant (J/(mol·K))
12    pub gas_constant: f64,
13    /// Temperature in Kelvin
14    pub temperature: f64,
15    /// Temperature in Celsius
16    pub temperature_celsius: f64,
17    /// RT/F at current temperature (mV)
18    pub rt_over_f: f64,
19}
20
21impl PhysicalConstants {
22    /// Create physical constants at a given temperature.
23    ///
24    /// # Arguments
25    ///
26    /// * `temp_celsius` - Temperature in degrees Celsius
27    ///
28    /// # Examples
29    ///
30    /// ```
31    /// use hodgkin_huxley::constants::PhysicalConstants;
32    ///
33    /// let constants = PhysicalConstants::new(6.3);
34    /// assert!((constants.temperature - 279.45).abs() < 0.1);
35    /// ```
36    pub fn new(temp_celsius: f64) -> Self {
37        const FARADAY: f64 = 96485.332; // C/mol
38        const R: f64 = 8.314; // J/(mol·K)
39
40        let temp_kelvin = temp_celsius + 273.15;
41        let rt_over_f = (R * temp_kelvin) / FARADAY * 1000.0; // Convert to mV
42
43        Self {
44            faraday: FARADAY,
45            gas_constant: R,
46            temperature: temp_kelvin,
47            temperature_celsius: temp_celsius,
48            rt_over_f,
49        }
50    }
51
52    /// Standard temperature (6.3°C, as used in original HH experiments).
53    pub fn standard() -> Self {
54        Self::new(6.3)
55    }
56
57    /// Physiological temperature (37°C).
58    pub fn physiological() -> Self {
59        Self::new(37.0)
60    }
61
62    /// Calculate Q10 temperature scaling factor.
63    ///
64    /// # Arguments
65    ///
66    /// * `q10` - Q10 value (typically 2-3 for ion channels)
67    /// * `reference_temp` - Reference temperature in Celsius
68    ///
69    /// # Examples
70    ///
71    /// ```
72    /// use hodgkin_huxley::constants::PhysicalConstants;
73    ///
74    /// let constants = PhysicalConstants::new(37.0);
75    /// let scale = constants.q10_factor(3.0, 6.3);
76    /// assert!(scale > 1.0); // Higher temperature increases rate
77    /// ```
78    pub fn q10_factor(&self, q10: f64, reference_temp: f64) -> f64 {
79        q10.powf((self.temperature_celsius - reference_temp) / 10.0)
80    }
81}
82
83impl Default for PhysicalConstants {
84    fn default() -> Self {
85        Self::standard()
86    }
87}

89/// Ion concentrations inside and outside the cell.
90///
91/// All concentrations in mM (millimolar).
92#[derive(Debug, Clone, Copy)]
93pub struct IonConcentrations {
94    /// Intracellular Na+ concentration (mM)
95    pub na_in: f64,
96    /// Extracellular Na+ concentration (mM)
97    pub na_out: f64,
98    /// Intracellular K+ concentration (mM)
99    pub k_in: f64,
100    /// Extracellular K+ concentration (mM)
101    pub k_out: f64,
102    /// Intracellular Ca2+ concentration (mM)
103    pub ca_in: f64,
104    /// Extracellular Ca2+ concentration (mM)
105    pub ca_out: f64,
106    /// Intracellular Cl- concentration (mM)
107    pub cl_in: f64,
108    /// Extracellular Cl- concentration (mM)
109    pub cl_out: f64,
110}
111
112impl IonConcentrations {
113    /// Create ion concentrations with custom values.
114    pub fn new(
115        na_in: f64, na_out: f64,
116        k_in: f64, k_out: f64,
117        ca_in: f64, ca_out: f64,
118        cl_in: f64, cl_out: f64,
119    ) -> Self {
120        Self {
121            na_in, na_out,
122            k_in, k_out,
123            ca_in, ca_out,
124            cl_in, cl_out,
125        }
126    }
127
128    /// Standard squid axon concentrations (from Hodgkin & Huxley 1952).
129    pub fn squid_axon() -> Self {
130        Self {
131            na_in: 50.0,
132            na_out: 437.0,
133            k_in: 400.0,
134            k_out: 20.0,
135            ca_in: 0.0001,
136            ca_out: 2.0,
137            cl_in: 52.0,
138            cl_out: 560.0,
139        }
140    }
141
142    /// Mammalian neuron concentrations (physiological).
143    pub fn mammalian() -> Self {
144        Self {
145            na_in: 15.0,
146            na_out: 145.0,
147            k_in: 140.0,
148            k_out: 5.0,
149            ca_in: 0.0001,
150            ca_out: 2.0,
151            cl_in: 10.0,
152            cl_out: 120.0,
153        }
154    }
155
156    /// Calculate Nernst potential for an ion.
157    ///
158    /// # Arguments
159    ///
160    /// * `conc_in` - Intracellular concentration (mM)
161    /// * `conc_out` - Extracellular concentration (mM)
162    /// * `valence` - Ion valence (e.g., +1 for Na+, +2 for Ca2+, -1 for Cl-)
163    /// * `constants` - Physical constants
164    ///
165    /// # Returns
166    ///
167    /// Nernst potential in mV
168    ///
169    /// # Examples
170    ///
171    /// ```
172    /// use hodgkin_huxley::constants::{IonConcentrations, PhysicalConstants};
173    ///
174    /// let conc = IonConcentrations::squid_axon();
175    /// let phys = PhysicalConstants::standard();
176    /// let e_na = IonConcentrations::nernst(conc.na_in, conc.na_out, 1.0, &phys);
177    /// assert!(e_na > 40.0 && e_na < 60.0); // Typical Na+ reversal potential
178    /// ```
179    pub fn nernst(conc_in: f64, conc_out: f64, valence: f64, constants: &PhysicalConstants) -> f64 {
180        (constants.rt_over_f / valence) * (conc_out / conc_in).ln()
181    }
182
183    /// Get sodium Nernst potential.
184    pub fn e_na(&self, constants: &PhysicalConstants) -> f64 {
185        Self::nernst(self.na_in, self.na_out, 1.0, constants)
186    }
187
188    /// Get potassium Nernst potential.
189    pub fn e_k(&self, constants: &PhysicalConstants) -> f64 {
190        Self::nernst(self.k_in, self.k_out, 1.0, constants)
191    }
192
193    /// Get calcium Nernst potential.
194    pub fn e_ca(&self, constants: &PhysicalConstants) -> f64 {
195        Self::nernst(self.ca_in, self.ca_out, 2.0, constants)
196    }
197
198    /// Get chloride Nernst potential.
199    pub fn e_cl(&self, constants: &PhysicalConstants) -> f64 {
200        Self::nernst(self.cl_in, self.cl_out, -1.0, constants)
201    }
202}
203
204impl Default for IonConcentrations {
205    fn default() -> Self {
206        Self::squid_axon()
207    }
208}

210#[cfg(test)]
211mod tests {
212    use super::*;
213
214    #[test]
215    fn test_physical_constants() {
216        let constants = PhysicalConstants::new(6.3);
217        assert!((constants.temperature - 279.45).abs() < 0.1);
218        // RT/F at 6.3°C should be around 23.26 mV
219        // R = 8.314 J/(mol·K), F = 96485.332 C/mol
220        // RT/F = (8.314 * 279.45) / 96485.332 * 1000 ≈ 24.04 mV
221        assert!((constants.rt_over_f - 24.04).abs() < 0.5);
222    }
223
224    #[test]
225    fn test_q10_scaling() {
226        let constants = PhysicalConstants::new(37.0);
227        let scale = constants.q10_factor(3.0, 6.3);
228        // Should be approximately 3^3.07 ≈ 27.6
229        assert!(scale > 20.0 && scale < 35.0);
230    }
231
232    #[test]
233    fn test_nernst_potential() {
234        let conc = IonConcentrations::squid_axon();
235        let phys = PhysicalConstants::standard();
236
237        let e_na = conc.e_na(&phys);
238        let e_k = conc.e_k(&phys);
239
240        // Na+ should be positive (higher outside)
241        assert!(e_na > 40.0);
242        // K+ should be negative (higher inside)
243        assert!(e_k < -60.0);
244    }
245
246    #[test]
247    fn test_mammalian_concentrations() {
248        let conc = IonConcentrations::mammalian();
249        let phys = PhysicalConstants::physiological();
250
251        let e_na = conc.e_na(&phys);
252        let e_k = conc.e_k(&phys);
253
254        // Typical mammalian values
255        assert!(e_na > 50.0 && e_na < 70.0);
256        assert!(e_k > -100.0 && e_k < -70.0);
257    }
