1//! # Hodgkin-Huxley Neuron Library
2//!
3//! A production-ready implementation of the Hodgkin-Huxley neuron model with exact
4//! biophysical equations from the seminal 1952 paper.
5//!
6//! ## Overview
7//!
8//! This library provides a complete implementation of the Hodgkin-Huxley equations
9//! for modeling action potentials in neurons. It includes:
10//!
11//! - **Accurate biophysics**: Exact equations from Hodgkin & Huxley (1952)
12//! - **Multiple neuron types**: Regular spiking, fast spiking, intrinsically bursting, etc.
13//! - **Numerical solvers**: RK4 and exponential Euler integrators
14//! - **Temperature effects**: Q10 scaling for realistic temperature dependence
15//! - **Ion channels**: Na⁺, K⁺, Ca²⁺-activated K⁺, and leak channels
16//!
17//! ## Mathematical Model
18//!
19//! The membrane voltage is governed by:
20//!
21//! ```text
22//! C_m * dV/dt = -I_Na - I_K - I_K(Ca) - I_leak + I_ext
23//! ```
24//!
25//! Where:
26//! - `I_Na = g_Na * m³ * h * (V - E_Na)` - Fast sodium current
27//! - `I_K = g_K * n⁴ * (V - E_K)` - Delayed rectifier potassium current
28//! - `I_K(Ca) = g_K(Ca) * a * b * (V - E_K)` - Calcium-activated potassium current
29//! - `I_leak = g_leak * (V - E_leak)` - Leak current
30//!
31//! Gating variables evolve according to:
32//!
33//! ```text
34//! dm/dt = α_m(V) * (1 - m) - β_m(V) * m
35//! dh/dt = α_h(V) * (1 - h) - β_h(V) * h
36//! dn/dt = α_n(V) * (1 - n) - β_n(V) * n
37//! da/dt = α_a(V) * (1 - a) - β_a(V) * a
38//! db/dt = α_b(V) * (1 - b) - β_b(V) * b
39//! ```
40//!
41//! ## Quick Start
42//!
43//! ```
44//! use hodgkin_huxley::{HodgkinHuxleyNeuron, neuron_types::NeuronConfig};
45//! use nalgebra::DVector;
46//!
47//! // Create a regular spiking cortical neuron
48//! let config = NeuronConfig::regular_spiking();
49//! let mut neuron = HodgkinHuxleyNeuron::new(config).unwrap();
50//!
51//! // Initialize at resting state
52//! neuron.initialize_rest();
53//!
54//! // Apply a current pulse and simulate
55//! let i_ext = 10.0; // µA/cm²
56//! let dt = 0.01;    // ms
57//! let duration = 50.0; // ms
58//!
59//! for _ in 0..(duration / dt) as usize {
60//!     neuron.step(dt, i_ext).unwrap();
61//! }
62//!
63//! // Check if neuron spiked
64//! let spikes = neuron.detect_spikes(-20.0);
65//! println!("Number of spikes: {}", spikes.len());
66//! ```
67//!
68//! ## Examples
69//!
70//! ### Simulate different neuron types
71//!
72//! ```
73//! use hodgkin_huxley::{HodgkinHuxleyNeuron, neuron_types::NeuronConfig};
74//!
75//! // Fast spiking interneuron
76//! let fs_neuron = HodgkinHuxleyNeuron::new(NeuronConfig::fast_spiking()).unwrap();
77//!
78//! // Intrinsically bursting neuron
79//! let ib_neuron = HodgkinHuxleyNeuron::new(NeuronConfig::intrinsically_bursting()).unwrap();
80//!
81//! // Classical squid axon
82//! let squid_neuron = HodgkinHuxleyNeuron::new(NeuronConfig::squid_axon()).unwrap();
83//! ```
84//!
85//! ### Record voltage trace
86//!
87//! ```
88//! use hodgkin_huxley::{HodgkinHuxleyNeuron, neuron_types::NeuronConfig};
89//!
90//! let mut neuron = HodgkinHuxleyNeuron::new(NeuronConfig::regular_spiking()).unwrap();
91//! neuron.initialize_rest();
92//!
93//! let mut voltage_trace = Vec::new();
94//! let dt = 0.01;
95//!
96//! for t in 0..5000 {
97//!     let current = if t > 1000 && t < 4000 { 10.0 } else { 0.0 };
98//!     neuron.step(dt, current).unwrap();
99//!     voltage_trace.push((t as f64 * dt, neuron.voltage()));
100//! }
101//! ```

116/// Complete Hodgkin-Huxley neuron model with 6 state variables.
117///
118/// State vector: [V, m, h, n, a, b]
119/// - V: Membrane potential (mV)
120/// - m: Sodium activation gate
121/// - h: Sodium inactivation gate
122/// - n: Potassium activation gate
123/// - a: Calcium-activated potassium activation gate
124/// - b: Calcium-activated potassium inactivation gate
125#[derive(Debug, Clone)]
126pub struct HodgkinHuxleyNeuron {
127    /// Current state vector
128    state: DVector<f64>,
129    /// Membrane capacitance (µF/cm²)
130    c_m: f64,
131    /// Sodium channel
132    na_channel: SodiumChannel,
133    /// Potassium channel
134    k_channel: PotassiumChannel,
135    /// Calcium-activated potassium channel
136    k_ca_channel: CalciumPotassiumChannel,
137    /// Leak channel
138    leak_channel: LeakChannel,
139    /// Ion concentrations
140    concentrations: IonConcentrations,
141    /// Physical constants
142    constants: PhysicalConstants,
143    /// Reversal potentials (cached)
144    e_na: f64,
145    e_k: f64,
146    /// Current simulation time (ms)
147    time: f64,
148    /// Voltage history for spike detection
149    voltage_history: Vec<(f64, f64)>,
150    /// Integration method
151    integrator: RK4Integrator,
152}

155    /// Create a new Hodgkin-Huxley neuron from configuration.
156    ///
157    /// # Arguments
158    ///
159    /// * `config` - Neuron configuration
160    ///
161    /// # Examples
162    ///
163    /// ```
164    /// use hodgkin_huxley::{HodgkinHuxleyNeuron, neuron_types::NeuronConfig};
165    ///
166    /// let neuron = HodgkinHuxleyNeuron::new(NeuronConfig::squid_axon()).unwrap();
167    /// ```
168    pub fn new(config: NeuronConfig) -> Result<Self> {
169        if config.c_m <= 0.0 {
170            return Err(HHError::InvalidParameter {
171                parameter: "c_m".to_string(),
172                value: config.c_m,
173                reason: "Capacitance must be positive".to_string(),
174            });
175        }
176
177        let e_na = config.concentrations.e_na(&config.constants);
178        let e_k = config.concentrations.e_k(&config.constants);
179
180        // Initialize state at resting values
181        let v_init = -65.0;
182        let state = DVector::from_vec(vec![
183            v_init,
184            SodiumChannel::m_inf(v_init),
185            SodiumChannel::h_inf(v_init),
186            PotassiumChannel::n_inf(v_init),
187            CalciumPotassiumChannel::a_inf(v_init),
188            CalciumPotassiumChannel::b_inf(v_init),
189        ]);
190
191        Ok(Self {
192            state,
193            c_m: config.c_m,
194            na_channel: config.na_channel,
195            k_channel: config.k_channel,
196            k_ca_channel: config.k_ca_channel,
197            leak_channel: config.leak_channel,
198            concentrations: config.concentrations,
199            constants: config.constants,
200            e_na,
201            e_k,
202            time: 0.0,
203            voltage_history: Vec::new(),
204            integrator: RK4Integrator::new(0.01)?,
205        })
206    }

228    /// Initialize neuron at resting state.
229    ///
230    /// Sets all gating variables to their steady-state values at the
231    /// resting potential.
232    pub fn initialize_rest(&mut self) {
233        let v_rest = self.leak_channel.e_leak;
234        self.state[0] = v_rest;
235        self.state[1] = SodiumChannel::m_inf(v_rest);
236        self.state[2] = SodiumChannel::h_inf(v_rest);
237        self.state[3] = PotassiumChannel::n_inf(v_rest);
238        self.state[4] = CalciumPotassiumChannel::a_inf(v_rest);
239        self.state[5] = CalciumPotassiumChannel::b_inf(v_rest);
240        self.time = 0.0;
241        self.voltage_history.clear();
242    }

244    /// Set membrane voltage.
245    pub fn set_voltage(&mut self, v: f64) -> Result<()> {
246        if !v.is_finite() {
247            return Err(HHError::NonFiniteValue {
248                location: "voltage".to_string(),
249                value: v,
250            });
251        }
252        self.state[0] = v;
253        Ok(())
254    }

256    /// Set gating variables.
257    pub fn set_gates(&mut self, m: f64, h: f64, n: f64, a: f64, b: f64) -> Result<()> {
258        let gates = [m, h, n, a, b];
259        let names = ["m", "h", "n", "a", "b"];
260
261        for (i, (&gate, &name)) in gates.iter().zip(names.iter()).enumerate() {
262            if !gate.is_finite() || gate < 0.0 || gate > 1.0 {
263                return Err(HHError::InvalidParameter {
264                    parameter: name.to_string(),
265                    value: gate,
266                    reason: "Gating variable must be in [0, 1]".to_string(),
267                });
268            }
269            self.state[i + 1] = gate;
270        }
271        Ok(())
272    }

300    /// Compute all ionic currents at current state.
301    ///
302    /// # Returns
303    ///
304    /// Tuple of (I_Na, I_K, I_K_Ca, I_leak) in µA/cm²
305    pub fn currents(&self) -> (f64, f64, f64, f64) {
306        let v = self.state[0];
307        let (m, h, n, a, b) = self.gates();
308
309        let i_na = self.na_channel.current(v, m, h, self.e_na);
310        let i_k = self.k_channel.current(v, n, self.e_k);
311        let i_k_ca = self.k_ca_channel.current(v, a, b, self.e_k);
312        let i_leak = self.leak_channel.current(v);
313
314        (i_na, i_k, i_k_ca, i_leak)
315    }

317    /// Compute total membrane current.
318    ///
319    /// # Arguments
320    ///
321    /// * `i_ext` - External applied current (µA/cm²)
322    ///
323    /// # Returns
324    ///
325    /// Total current in µA/cm²
326    pub fn total_current(&self, i_ext: f64) -> f64 {
327        let (i_na, i_k, i_k_ca, i_leak) = self.currents();
328        i_na + i_k + i_k_ca + i_leak - i_ext
329    }

336    pub fn conductances(&self) -> (f64, f64, f64, f64) {

347    /// Compute derivatives of state variables (dY/dt).
348    ///
349    /// # Arguments
350    ///
351    /// * `i_ext` - External applied current (µA/cm²)
352    ///
353    /// # Returns
354    ///
355    /// Vector of derivatives [dV/dt, dm/dt, dh/dt, dn/dt, da/dt, db/dt]
356    fn derivatives(&self, i_ext: f64) -> DVector<f64> {
357        let v = self.state[0];
358        let (m, h, n, a, b) = self.gates();
359
360        // Ionic currents
361        let (i_na, i_k, i_k_ca, i_leak) = self.currents();
362
363        // dV/dt = (I_ext - I_Na - I_K - I_K_Ca - I_leak) / C_m
364        let dv_dt = (i_ext - i_na - i_k - i_k_ca - i_leak) / self.c_m;
365
366        // Gating variable rate constants
367        let alpha_m = SodiumChannel::alpha_m(v);
368        let beta_m = SodiumChannel::beta_m(v);
369        let alpha_h = SodiumChannel::alpha_h(v);
370        let beta_h = SodiumChannel::beta_h(v);
371        let alpha_n = PotassiumChannel::alpha_n(v);
372        let beta_n = PotassiumChannel::beta_n(v);
373        let alpha_a = CalciumPotassiumChannel::alpha_a(v);
374        let beta_a = CalciumPotassiumChannel::beta_a(v);
375        let alpha_b = CalciumPotassiumChannel::alpha_b(v);
376        let beta_b = CalciumPotassiumChannel::beta_b(v);
377
378        // Gating variable derivatives
379        let dm_dt = alpha_m * (1.0 - m) - beta_m * m;
380        let dh_dt = alpha_h * (1.0 - h) - beta_h * h;
381        let dn_dt = alpha_n * (1.0 - n) - beta_n * n;
382        let da_dt = alpha_a * (1.0 - a) - beta_a * a;
383        let db_dt = alpha_b * (1.0 - b) - beta_b * b;
384
385        DVector::from_vec(vec![dv_dt, dm_dt, dh_dt, dn_dt, da_dt, db_dt])
386    }

388    /// Advance simulation by one time step using RK4 integration.
389    ///
390    /// # Arguments
391    ///
392    /// * `dt` - Time step (ms)
393    /// * `i_ext` - External applied current (µA/cm²)
394    ///
395    /// # Examples
396    ///
397    /// ```
398    /// use hodgkin_huxley::{HodgkinHuxleyNeuron, neuron_types::NeuronConfig};
399    ///
400    /// let mut neuron = HodgkinHuxleyNeuron::squid_axon().unwrap();
401    /// neuron.initialize_rest();
402    ///
403    /// // Apply 10 µA/cm² current for 1 ms
404    /// neuron.step(0.01, 10.0).unwrap();
405    /// ```
406    pub fn step(&mut self, dt: f64, i_ext: f64) -> Result<()> {
407        // Update integrator if needed
408        if (dt - self.integrator.dt).abs() > 1e-10 {
409            self.integrator = RK4Integrator::new(dt)?;
410        }
411
412        // Save neuron parameters for closure
413        let c_m = self.c_m;
414        let na_channel = self.na_channel;
415        let k_channel = self.k_channel;
416        let k_ca_channel = self.k_ca_channel;
417        let leak_channel = self.leak_channel;
418        let concentrations = self.concentrations;
419        let constants = self.constants;
420        let e_na = self.e_na;
421        let e_k = self.e_k;
422
423        // Define ODE function with captured parameters
424        let f = move |_t: f64, y: &DVector<f64>| -> DVector<f64> {
425            let temp_neuron = HodgkinHuxleyNeuron {
426                state: y.clone(),
427                c_m,
428                na_channel,
429                k_channel,
430                k_ca_channel,
431                leak_channel,
432                concentrations,
433                constants,
434                e_na,
435                e_k,
436                time: 0.0,
437                voltage_history: Vec::new(),
438                integrator: RK4Integrator::new(0.01).unwrap(),
439            };
440            temp_neuron.derivatives(i_ext)
441        };
442
443        // Integrate
444        let current_time = self.time;
445        let current_state = self.state.clone();
446        self.state = self.integrator.step(current_time, &current_state, &f)?;
447        self.time += dt;
448
449        // Record voltage for spike detection
450        self.voltage_history.push((self.time, self.voltage()));
451
452        // Keep only recent history (last 100 ms)
453        if let Some(&(first_time, _)) = self.voltage_history.first() {
454            if self.time - first_time > 100.0 {
455                let cutoff_time = self.time - 100.0;
456                self.voltage_history.retain(|(t, _)| *t >= cutoff_time);
457            }
458        }
459
460        Ok(())
461    }

463    /// Simulate for a duration with constant external current.
464    ///
465    /// # Arguments
466    ///
467    /// * `duration` - Simulation duration (ms)
468    /// * `dt` - Time step (ms)
469    /// * `i_ext` - External applied current (µA/cm²)
470    ///
471    /// # Returns
472    ///
473    /// Vector of (time, voltage) pairs
474    pub fn simulate(&mut self, duration: f64, dt: f64, i_ext: f64) -> Result<Vec<(f64, f64)>> {
475        let n_steps = (duration / dt).ceil() as usize;
476        let mut trace = Vec::with_capacity(n_steps);
477
478        for _ in 0..n_steps {
479            self.step(dt, i_ext)?;
480            trace.push((self.time, self.voltage()));
481        }
482
483        Ok(trace)
484    }

486    /// Detect action potential spikes in voltage history.
487    ///
488    /// # Arguments
489    ///
490    /// * `threshold` - Voltage threshold for spike detection (mV), typically -20 mV
491    ///
492    /// # Returns
493    ///
494    /// Vector of spike times (ms)
495    ///
496    /// # Examples
497    ///
498    /// ```
499    /// use hodgkin_huxley::{HodgkinHuxleyNeuron, neuron_types::NeuronConfig};
500    ///
501    /// let mut neuron = HodgkinHuxleyNeuron::regular_spiking().unwrap();
502    /// neuron.initialize_rest();
503    /// neuron.simulate(100.0, 0.01, 10.0).unwrap();
504    ///
505    /// let spikes = neuron.detect_spikes(-20.0);
506    /// println!("Detected {} spikes", spikes.len());
507    /// ```
508    pub fn detect_spikes(&self, threshold: f64) -> Vec<f64> {
509        let mut spikes = Vec::new();
510        let mut above_threshold = false;
511
512        for window in self.voltage_history.windows(2) {
513            let (t1, v1) = window[0];
514            let (_, v2) = window[1];
515
516            // Detect upward threshold crossing
517            if !above_threshold && v1 <= threshold && v2 > threshold {
518                spikes.push(t1);
519                above_threshold = true;
520            }
521
522            // Reset when falling below threshold
523            if above_threshold && v2 <= threshold {
524                above_threshold = false;
525            }
526        }
527
528        spikes
529    }

531    /// Calculate instantaneous firing rate from spike times.
532    ///
533    /// # Arguments
534    ///
535    /// * `spikes` - Vector of spike times (ms)
536    ///
537    /// # Returns
538    ///
539    /// Average firing rate in Hz
540    pub fn firing_rate(spikes: &[f64]) -> f64 {
541        if spikes.len() < 2 {
542            return 0.0;
543        }
544
545        let duration = spikes[spikes.len() - 1] - spikes[0];
546        if duration <= 0.0 {
547            return 0.0;
548        }
549
550        ((spikes.len() - 1) as f64 / duration) * 1000.0 // Convert ms to s
551    }

553    /// Calculate interspike intervals.
554    ///
555    /// # Arguments
556    ///
557    /// * `spikes` - Vector of spike times (ms)
558    ///
559    /// # Returns
560    ///
561    /// Vector of interspike intervals (ms)
562    pub fn interspike_intervals(spikes: &[f64]) -> Vec<f64> {
563        if spikes.len() < 2 {
564            return Vec::new();
565        }
566
567        spikes.windows(2)
568            .map(|w| w[1] - w[0])
569            .collect()
570    }

647        // but they balance out. The leak reversal potential determines the exact state.
648        // With E_leak = -54.387 mV, there will be steady-state currents.
649        assert!(i_na.abs() < 50.0, "i_na = {} exceeded bounds", i_na);
650        assert!(i_k.abs() < 50.0, "i_k = {} exceeded bounds", i_k);
651        assert!(i_k_ca.abs() < 1.0, "i_k_ca = {} exceeded bounds", i_k_ca);
652        assert!(i_leak.abs() < 50.0, "i_leak = {} exceeded bounds", i_leak);
653
654        // All values should be finite
655        assert!(i_na.is_finite());
656        assert!(i_k.is_finite());
657        assert!(i_k_ca.is_finite());
658        assert!(i_leak.is_finite());
659    }
